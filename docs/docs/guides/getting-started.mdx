# Getting Started

This guide walks you through installing Flipova Foundation and building your first screen — on React Native **or** React Web.

## Installation

```bash
npm install @flipova/foundation
```

:::info Registry
The package is published on the **npm public registry**. No extra `.npmrc` configuration is needed.
:::

### Peer dependencies (React Native)

Install the native peer deps you need. All are optional — Foundation detects availability at runtime:

```bash
npx expo install \
  expo-linear-gradient \
  expo-haptics \
  expo-status-bar \
  react-native-gesture-handler \
  react-native-reanimated \
  react-native-safe-area-context \
  react-native-screens \
  lucide-react-native
```

### For web projects (Vite, Next.js…)

No peer deps needed. Import from the `/web` sub-entry to get a pure DOM bundle:

```bash
npm install @flipova/foundation
# that's it — no react-native required
```

## React Native quick start

### 1. Wrap your app

```tsx title="App.tsx"
import { FoundationProvider, defineConfig } from '@flipova/foundation/config';

const config = defineConfig({ defaultTheme: 'light' });

export default function App() {
  return (
    <FoundationProvider config={config}>
      <RootNavigator />
    </FoundationProvider>
  );
}
```

### 2. Use layout primitives

```tsx title="HomeScreen.tsx"
import { Box, Stack, Inline } from '@flipova/foundation';
import { Button, Avatar, Badge } from '@flipova/foundation';

export default function HomeScreen() {
  return (
    <Box p={5} bg="background" style={{ flex: 1 }}>
      <Stack spacing={4}>
        <Inline spacing={3} alignItems="center">
          <Avatar source="https://i.pravatar.cc/80" size="lg" />
          <Stack spacing={1}>
            <Box><Badge label="Pro" color="primary" /></Box>
          </Stack>
        </Inline>
        <Button
          variant="primary"
          label="Get started"
          onPress={() => {}}
          fullWidth
        />
      </Stack>
    </Box>
  );
}
```

## React Web quick start

### 1. Wrap your app

```tsx title="App.tsx"
import { ThemeProvider } from '@flipova/foundation/web';

export default function App() {
  return (
    <ThemeProvider defaultTheme="light">
      <Router />
    </ThemeProvider>
  );
}
```

### 2. Use web components

```tsx title="HomePage.tsx"
import { Box, Stack, Button, Avatar } from '@flipova/foundation/web';

export default function HomePage() {
  return (
    <Box p={5} borderRadius="lg" style={{ maxWidth: 480, margin: '0 auto' }}>
      <Stack spacing={4}>
        <Avatar source="https://i.pravatar.cc/80" size="lg" />
        <Button variant="primary">Get started</Button>
      </Stack>
    </Box>
  );
}
```

:::tip Vite alias
If you see `Cannot resolve "react-native"` errors, make sure you're importing from `@flipova/foundation/web` (not the main entry) in browser projects. The web sub-module is built without any React Native references.
:::

## Custom configuration

### Override tokens

```ts title="flipova.config.ts"
import { defineConfig } from '@flipova/foundation/config';

export default defineConfig({
  tokens: {
    spacing: { 0:0, 1:4, 2:8, 3:12, 4:16, 5:20, 6:24, 8:32 },
    radii:   { none:0, sm:4, base:6, md:8, lg:12, xl:16, '2xl':20, full:9999 },
  },
  defaultTheme: 'light',
});
```

## Flipova Studio

Generate entire React Native screens visually:

```bash
npx flipova-studio
# Opens http://localhost:4200
```

Or run on a custom port:

```bash
npx flipova-studio --port 3000
```

Generate code to `./generated/`:

```bash
npx flipova-studio generate
```

See the [Studio guide](./studio) for full documentation.

## TypeScript support

Foundation ships full type definitions. You get autocomplete on every prop:

```tsx
import type { ButtonProps, AvatarProps } from '@flipova/foundation/web';

const btnProps: ButtonProps = {
  variant: 'primary',   // ✅  'primary' | 'secondary' | 'outline' | 'ghost' | 'destructive'
  size: 'md',           // ✅  'sm' | 'md' | 'lg'
};
```
